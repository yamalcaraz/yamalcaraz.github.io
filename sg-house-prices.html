In [165]:
import numpy as np

def ecdf(data):
    """Compute Empirical Cumulative Distribution Function for a one-dimensional array of measurements."""

    # Number of data points: n
    n=len(data)

    # x-data for the ECDF: x
    x=np.sort(data)

    # y-data for the ECDF: y
    y = np.arange(1, n+1) / n

    return x, y
